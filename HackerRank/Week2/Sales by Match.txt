#include <stdio.h>

int main() {
    int n;
    scanf("%d", &n);
    int arr[n];
    
    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }
    
    int count[101] = {0};  // assuming sock colors are in [1..100]
    for (int i = 0; i < n; i++) {
        count[arr[i]]++;
    }
    
    int pairs = 0;
    for (int i = 0; i < 101; i++) {
        pairs += count[i] / 2;  // each two socks form one pair
    }
    
    printf("%d\n", pairs);
    return 0;
}
