#include <stdio.h>
#include <stdlib.h>

int main() {
    int n, q;
    scanf("%d %d", &n, &q);

    // Allocate n sequences dynamically
    int **seqList = (int **)malloc(n * sizeof(int *));
    int *sizes = (int *)calloc(n, sizeof(int));   // store current sizes
    int *capacity = (int *)malloc(n * sizeof(int)); // store capacities

    for (int i = 0; i < n; i++) {
        seqList[i] = NULL;
        sizes[i] = 0;
        capacity[i] = 0;
    }

    int lastAnswer = 0;

    for (int i = 0; i < q; i++) {
        int type, x, y;
        scanf("%d %d %d", &type, &x, &y);

        int seq = (x ^ lastAnswer) % n;

        if (type == 1) {
            // expand capacity if needed
            if (sizes[seq] == capacity[seq]) {
                capacity[seq] = (capacity[seq] == 0) ? 1 : capacity[seq] * 2;
                seqList[seq] = realloc(seqList[seq], capacity[seq] * sizeof(int));
            }
            seqList[seq][sizes[seq]++] = y;

        } else if (type == 2) {
            int idx = y % sizes[seq];
            lastAnswer = seqList[seq][idx];
            printf("%d\n", lastAnswer);
        }
    }

    // Free memory
    for (int i = 0; i < n; i++) {
        free(seqList[i]);
    }
    free(seqList);
    free(sizes);
    free(capacity);

    return 0;
}
