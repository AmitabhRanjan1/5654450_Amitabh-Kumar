#include <stdio.h>

int main() {
    int t;
    scanf("%d", &t);

    while (t--) {
        int m, n;
        scanf("%d %d", &m, &n);

        int cost[n];
        for (int i = 0; i < n; i++) {
            scanf("%d", &cost[i]);
        }

        // Find the pair
        for (int i = 0; i < n - 1; i++) {
            for (int j = i + 1; j < n; j++) {
                if (cost[i] + cost[j] == m) {
                    printf("%d %d\n", i + 1, j + 1);
                    goto found;  // break out of both loops
                }
            }
        }
        found: ;
    }
    return 0;
}
