#include <stdio.h>
#include <stdlib.h>

int main() {
    int n, m;
    scanf("%d", &n);

    int scores[n];
    for (int i = 0; i < n; i++) {
        scanf("%d", &scores[i]);
    }

    scanf("%d", &m);
    int alice[m];
    for (int i = 0; i < m; i++) {
        scanf("%d", &alice[i]);
    }

    // Step 1: Build unique ranked scores
    int *unique = (int *)malloc(n * sizeof(int));
    int size = 0;
    unique[size++] = scores[0];
    for (int i = 1; i < n; i++) {
        if (scores[i] != scores[i - 1]) {
            unique[size++] = scores[i];
        }
    }

    // Step 2: Start from the bottom of leaderboard
    int idx = size - 1;
    for (int i = 0; i < m; i++) {
        int score = alice[i];
        while (idx >= 0 && score >= unique[idx]) {
            idx--;
        }
        printf("%d\n", idx + 2);
    }

    free(unique);
    return 0;
}
