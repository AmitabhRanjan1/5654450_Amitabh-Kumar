#include <stdio.h>
#include <stdlib.h> // for abs()

int main() {
    int n;
    scanf("%d", &n);

    int primary_sum = 0, secondary_sum = 0;
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            int val;
            scanf("%d", &val);
            if (i == j) {
                primary_sum += val; // primary diagonal
            }
            if (i + j == n - 1) {
                secondary_sum += val; // secondary diagonal
            }
        }
    }

    int difference = abs(primary_sum - secondary_sum);
    printf("%d\n", difference);

    return 0;
}
