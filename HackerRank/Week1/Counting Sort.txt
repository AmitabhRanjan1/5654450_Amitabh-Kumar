#include <stdio.h>

int main() {
    int n;
    scanf("%d", &n);

    int count[100] = {0}; // frequency array for numbers 0-99

    for (int i = 0; i < n; i++) {
        int val;
        scanf("%d", &val);
        count[val]++; // increment frequency
    }

    for (int i = 0; i < 100; i++) {
        printf("%d ", count[i]);
    }

    return 0;
}
