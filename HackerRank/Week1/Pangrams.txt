#include <stdio.h>
#include <ctype.h>  // for tolower()
#include <string.h> // for strlen()

int main() {
    char str[1005]; // buffer for input
    fgets(str, sizeof(str), stdin);

    int letters[26] = {0}; // track each letter's occurrence
    int count = 0;

    for (int i = 0; str[i] != '\0'; i++) {
        if (isalpha(str[i])) {
            int idx = tolower(str[i]) - 'a';
            if (letters[idx] == 0) {
                letters[idx] = 1;
                count++;
            }
        }
    }

    if (count == 26)
        printf("pangram\n");
    else
        printf("not pangram\n");

    return 0;
}
