#include <stdio.h>
#include <stdlib.h>

int cmp_asc(const void *a, const void *b) {
    return (*(int*)a - *(int*)b);
}

int cmp_desc(const void *a, const void *b) {
    return (*(int*)b - *(int*)a);
}

int main() {
    int t;
    scanf("%d", &t);

    while (t--) {
        int n, k;
        scanf("%d %d", &n, &k);

        int A[n], B[n];
        for (int i = 0; i < n; i++) scanf("%d", &A[i]);
        for (int i = 0; i < n; i++) scanf("%d", &B[i]);

        // Sort A in ascending order, B in descending order
        qsort(A, n, sizeof(int), cmp_asc);
        qsort(B, n, sizeof(int), cmp_desc);

        int possible = 1;
        for (int i = 0; i < n; i++) {
            if (A[i] + B[i] < k) {
                possible = 0;
                break;
            }
        }

        if (possible) printf("YES\n");
        else printf("NO\n");
    }

    return 0;
}
