#include <stdio.h>
#include <stdlib.h>

long long arrayManipulation(int n, int m) {
    long long *arr = (long long *)calloc(n + 2, sizeof(long long));
    if (arr == NULL) {
        return -1; /* memory allocation failed */
    }

    for (int i = 0; i < m; i++) {
        int a, b;
        long long k;
        scanf("%d %d %lld", &a, &b, &k);
        arr[a] += k;
        arr[b + 1] -= k;
    }

    long long max = 0, curr = 0;
    for (int i = 1; i <= n; i++) {
        curr += arr[i];
        if (curr > max) {
            max = curr;
        }
    }

    free(arr);
    return max;
}

int main() {
    int n, m;
    scanf("%d %d", &n, &m);

    long long result = arrayManipulation(n, m);
    printf("%lld\n", result);

    return 0;
}
