#include <stdio.h>

int main() {
    int n1, n2, n3;
    scanf("%d %d %d", &n1, &n2, &n3);

    int h1[n1], h2[n2], h3[n3];
    int sum1 = 0, sum2 = 0, sum3 = 0;

    // Read first stack
    for (int i = 0; i < n1; i++) {
        scanf("%d", &h1[i]);
        sum1 += h1[i];
    }
    // Read second stack
    for (int i = 0; i < n2; i++) {
        scanf("%d", &h2[i]);
        sum2 += h2[i];
    }
    // Read third stack
    for (int i = 0; i < n3; i++) {
        scanf("%d", &h3[i]);
        sum3 += h3[i];
    }

    // Pointers to top of each stack
    int i = 0, j = 0, k = 0;

    // Keep removing until equal
    while (!(sum1 == sum2 && sum2 == sum3)) {
        if (sum1 >= sum2 && sum1 >= sum3) {
            sum1 -= h1[i++];
        } else if (sum2 >= sum1 && sum2 >= sum3) {
            sum2 -= h2[j++];
        } else {
            sum3 -= h3[k++];
        }
    }

    printf("%d\n", sum1);
    return 0;
}
